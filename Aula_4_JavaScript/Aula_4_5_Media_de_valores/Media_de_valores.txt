####    Média de valores

* No arquivo JS, coloque isso, para ver na prática como funciona a média e como usar o "if" e "else".

    //  Média de valores

    // criando as variáveis com os valores das notas
    var nota1 = 8;
    var nota2 = 7;

    // realizando a operação de média das notas
    var media = ( nota1 + nota2 ) / 2;

    // Usando uma condição, o "if" que significa 'se' e o "else" que significa 'senão'

    // se a media for maior ou igual a 7, exibi a mensagem abaixo
    if( media >= 7 ){
        alert("Média: "+media+" Aluno foi aprovado! ")

    // se não, exibirá essa outra mensagem
    } else {
        alert("Média: "+media+" Aluno foi reprovado! ")
    }
